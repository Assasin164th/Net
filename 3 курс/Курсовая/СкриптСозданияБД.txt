DROP TABLE IF EXISTS SaleItems;
DROP TABLE IF EXISTS Sales;
DROP TABLE IF EXISTS PurchaseItems;
DROP TABLE IF EXISTS Purchases;
DROP TABLE IF EXISTS Catch;
DROP TABLE IF EXISTS TripCrew;
DROP TABLE IF EXISTS Trips;
DROP TABLE IF EXISTS FishingSpots;
DROP TABLE IF EXISTS CrewMembers;
DROP TABLE IF EXISTS Boats;
DROP TABLE IF EXISTS ProductImages;
DROP TABLE IF EXISTS Products;
DROP TABLE IF EXISTS Users;
DROP TABLE IF EXISTS Customers;
DROP TABLE IF EXISTS Suppliers;
DROP TABLE IF EXISTS Manufacturers;
DROP TABLE IF EXISTS Categories;
DROP TABLE IF EXISTS SiteImages;
DROP TABLE IF EXISTS Roles;
GO

CREATE TABLE Roles (
    RoleID INT PRIMARY KEY IDENTITY,
    RoleName NVARCHAR(50) NOT NULL UNIQUE
);
GO

CREATE TABLE Categories (
    CategoryID INT PRIMARY KEY IDENTITY,
    CategoryName NVARCHAR(100) NOT NULL,
    Description NVARCHAR(255) NULL,
    ParentCategoryID INT NULL,
    FOREIGN KEY (ParentCategoryID) REFERENCES Categories(CategoryID)
);
GO

CREATE TABLE Manufacturers (
    ManufacturerID INT PRIMARY KEY IDENTITY,
    Name NVARCHAR(150) NOT NULL,
    Country NVARCHAR(100) NULL
);
GO

CREATE TABLE Suppliers (
    SupplierID INT PRIMARY KEY IDENTITY,
    CompanyName NVARCHAR(200) NOT NULL,
    ContactPerson NVARCHAR(100) NULL,
    Phone NVARCHAR(20) NULL,
    Email NVARCHAR(100) NULL,
    Address NVARCHAR(255) NULL
);
GO

CREATE TABLE Customers (
    CustomerID INT PRIMARY KEY IDENTITY,
    FirstName NVARCHAR(50) NOT NULL,
    LastName NVARCHAR(50) NOT NULL,
    Phone NVARCHAR(20) NULL,
    Email NVARCHAR(100) NULL,
    DiscountCard BIT DEFAULT 0,
    RegistrationDate DATETIME DEFAULT GETDATE()
);
GO

CREATE TABLE Users (
    UserID INT PRIMARY KEY IDENTITY,
    FullName NVARCHAR(100) NOT NULL,
    Login NVARCHAR(50) NOT NULL UNIQUE,
    PasswordHash NVARCHAR(255) NOT NULL,
    RoleID INT NOT NULL,
    FOREIGN KEY (RoleID) REFERENCES Roles(RoleID)
);
GO

CREATE TABLE Products (
    ProductID INT PRIMARY KEY IDENTITY,
    ProductName NVARCHAR(255) NOT NULL,
    CategoryID INT NULL,
    ManufacturerID INT NULL,
    Description NVARCHAR(MAX) NULL,
    QuantityInStock INT NOT NULL DEFAULT 0,
    Unit NVARCHAR(20) NOT NULL,
    CostPrice DECIMAL(10,2) NULL,
    RetailPrice DECIMAL(10,2) NOT NULL,
    MinStock INT DEFAULT 0,
    FOREIGN KEY (CategoryID) REFERENCES Categories(CategoryID),
    FOREIGN KEY (ManufacturerID) REFERENCES Manufacturers(ManufacturerID)
);
GO

CREATE TABLE ProductImages (
    ImageID INT PRIMARY KEY IDENTITY,
    ProductID INT NOT NULL,
    ImageURL NVARCHAR(500) NOT NULL,
    IsMain BIT DEFAULT 0,
    SortOrder INT DEFAULT 0,
    AltText NVARCHAR(255) NULL,
    UploadedDate DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (ProductID) REFERENCES Products(ProductID)
);
GO

CREATE TABLE Boats (
    BoatID INT PRIMARY KEY IDENTITY,
    BoatName NVARCHAR(100) NOT NULL,
    BoatType NVARCHAR(50) NOT NULL,
    Displacement DECIMAL(10,2) NOT NULL,
    BuildDate DATE NOT NULL
);
GO

CREATE TABLE CrewMembers (
    CrewMemberID INT PRIMARY KEY IDENTITY,
    LastName NVARCHAR(50) NOT NULL,
    FirstName NVARCHAR(50) NOT NULL,
    Address NVARCHAR(255) NULL,
    Position NVARCHAR(50) NOT NULL
);
GO

CREATE TABLE FishingSpots (
    SpotID INT PRIMARY KEY IDENTITY,
    SpotName NVARCHAR(100) NOT NULL,
    Coordinates NVARCHAR(100) NULL
);
GO

CREATE TABLE Trips (
    TripID INT PRIMARY KEY IDENTITY,
    BoatID INT NOT NULL,
    DepartureDate DATE NOT NULL,
    ReturnDate DATE NULL,
    FOREIGN KEY (BoatID) REFERENCES Boats(BoatID)
);
GO

CREATE TABLE TripCrew (
    TripCrewID INT PRIMARY KEY IDENTITY,
    TripID INT NOT NULL,
    CrewMemberID INT NOT NULL,
    FOREIGN KEY (TripID) REFERENCES Trips(TripID),
    FOREIGN KEY (CrewMemberID) REFERENCES CrewMembers(CrewMemberID)
);
GO

CREATE TABLE Catch (
    CatchID INT PRIMARY KEY IDENTITY,
    TripID INT NOT NULL,
    SpotID INT NOT NULL,
    FishType NVARCHAR(50) NOT NULL,
    Weight DECIMAL(10,2) NOT NULL,
    Quality NVARCHAR(20) NOT NULL,
    ArrivalDate DATE NOT NULL,
    DepartureDate DATE NOT NULL,
    FOREIGN KEY (TripID) REFERENCES Trips(TripID),
    FOREIGN KEY (SpotID) REFERENCES FishingSpots(SpotID)
);
GO

CREATE TABLE Purchases (
    PurchaseID INT PRIMARY KEY IDENTITY,
    SupplierID INT NOT NULL,
    PurchaseDate DATETIME NOT NULL DEFAULT GETDATE(),
    TotalAmount DECIMAL(10,2) NULL,
    FOREIGN KEY (SupplierID) REFERENCES Suppliers(SupplierID)
);
GO

CREATE TABLE PurchaseItems (
    PurchaseItemID INT PRIMARY KEY IDENTITY,
    PurchaseID INT NOT NULL,
    ProductID INT NOT NULL,
    Quantity INT NOT NULL,
    PricePerUnit DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (PurchaseID) REFERENCES Purchases(PurchaseID),
    FOREIGN KEY (ProductID) REFERENCES Products(ProductID)
);
GO

CREATE TABLE Sales (
    SaleID INT PRIMARY KEY IDENTITY,
    CustomerID INT NULL,
    SaleDate DATETIME NOT NULL DEFAULT GETDATE(),
    TotalAmount DECIMAL(10,2) NOT NULL,
    PaymentMethod NVARCHAR(50) NULL,
    UserID INT NULL,
    FOREIGN KEY (CustomerID) REFERENCES Customers(CustomerID),
    FOREIGN KEY (UserID) REFERENCES Users(UserID)
);
GO

CREATE TABLE SaleItems (
    SaleItemID INT PRIMARY KEY IDENTITY,
    SaleID INT NOT NULL,
    ProductID INT NOT NULL,
    Quantity INT NOT NULL,
    PriceAtSale DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (SaleID) REFERENCES Sales(SaleID),
    FOREIGN KEY (ProductID) REFERENCES Products(ProductID)
);
GO

CREATE TABLE SiteImages (
    SiteImageID INT PRIMARY KEY IDENTITY,
    Title NVARCHAR(200) NULL,
    ImageURL NVARCHAR(500) NOT NULL,
    AltText NVARCHAR(255) NULL,
    Category NVARCHAR(50) NULL,
    Link NVARCHAR(500) NULL,
    SortOrder INT DEFAULT 0,
    UploadedDate DATETIME DEFAULT GETDATE()
);
GO
